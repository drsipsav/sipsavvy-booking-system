function getAddon(key) {
  const addons = JSON.parse(localStorage.getItem("ss_addons") || "{}");
  return Number(addons[key] || 0);
}

function getTravelFee() {
  return Number(localStorage.getItem("ss_travel_fee") || 0);
}
