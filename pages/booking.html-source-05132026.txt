<script>
  // Load selected package
  const pkg = localStorage.getItem('selectedPackage');
  if (pkg) {
    document.getElementById('selected-package').textContent = pkg;
  }

  // Load selected add-ons
  const addons = JSON.parse(localStorage.getItem('selectedAddons') || "[]");
  const addonsContainer = document.getElementById('selected-addons');

  if (addons.length) {
    addonsContainer.innerHTML = addons.map(a => `<li>${a}</li>`).join('');
  } else {
    addonsContainer.innerHTML = "<li>No add-ons selected</li>";
  }

  // Clear everything and return to packages
  document.getElementById('clearBooking').addEventListener('click', () => {
    localStorage.removeItem('selectedPackage');
    localStorage.removeItem('selectedAddons');
    window.location.href = "packages.html";
  });
</script>
